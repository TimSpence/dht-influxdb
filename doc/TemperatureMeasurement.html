def is_valid?
  scale == TemperatureScale::Celsius ?
    value > -7.0 && value < 51.0
  :
    value > 20.0 && value < 125.0
end

def to_celsius
  scale == TemperatureScale::Celsius ?
    value
  :
    (value - 32) * 5 / 9
end
